447

476          } else if (boardMatrix[2][0] == boardMatrix[2][2]) { 
477              Xpos = 2; 
478              Ypos = 2; 
479              return true; 
480          } 
481   
482          for (int i = 0; i < 3; i++) { 
483              for (int j = 0; j < 3; j++) { 
484   
485                  //System.out.println(board[i][j]); 
486   
487                  for (int k = i; k < 3; k++) { 
488                      for (int l = j; l < 3; l++) { 
489                          switch (j) { 
490                              case 0: { 
491                                  if ((k == i && l == j) || (k == i && l == j + 2) || (k == i + 1 && l == j + 2) || (k == i + 2)) 
492                                      continue; 
493                                  if (boardMatrix[i][j] == boardMatrix[k][l]) { 
494                                      Xpos = k; 
495                                      Ypos = l; 
496   
497                                      return true; 
498   
499                                  } 
500                              } 
501                              break; 
502                              case 1: { 
503                                  if (k == i && l == j) 
504                                      continue; 
505                                  if (boardMatrix[i][j] == boardMatrix[k][l]) { 
506                                      Xpos = k; 
507                                      Ypos = l; 
508                                      return true; 
509                                  } 
510                              } 
511                              break; 
512                              case 2: { 
513                                  if (k == i && l == j) 
514                                      continue; 
515                                  if (boardMatrix[i][j] == boardMatrix[k][l]) { 
516                                      Xpos = k; 
517                                      Ypos = l; 
518                                      return true; 
519                                  } 
520                              } 
521                              break; 
522                              default: 
523                                  break; 
524                          } 
525                      } 
526                  } 
527              } 
528          } 
529   
530          return false; 
531      }
